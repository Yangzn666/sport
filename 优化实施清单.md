function shareDocument(docName) {
    const url = `${window.location.origin}${window.location.pathname}#${docName}`;

}

// 监听hash
window.addEventListener('hashchange', () => {
    const docName = window.location.hash.substring(1);
    if (docName && knowledgeContents[docName]) {
        openViewer({ file: docName, title: docName });
    }
});
