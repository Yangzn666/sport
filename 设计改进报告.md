1. **学术权威性**
   - 衬线字体传递专业感
   - 无圆角设计体现严谨
   - 深色配色保持庄重
   - 双倍边框/双线分隔模仿期刊排版

2. **阅读舒适性**
   - 充足的留白（padding/margin）
   - 合理的行距（2.0-2.4）
   - 适当的字间距（0.5-2px）
   - 对比度控制（避免纯黑#000）

3. **交互自然性**
   - 缓动函数（cubic-bezier）
   - 微妙的位移（3-5px）
   - 渐进式阴影（多层叠加）
   - 状态反馈即时（hover/active）

```
第一层：页面背景（渐变灰 #f8f9fa → #e9ecef）
第二层：卡片容器（白色 + 阴影）
```

```javascript
// 简化版全文搜索算法
function searchKnowledge(query) {
    const results = [];
    const lowerQuery = query.toLowerCase();
    
    Object.keys(knowledgeContents).forEach(docName => {
        const content = knowledgeContents[docName];
        const lowerContent = content.toLowerCase();
        
        if (lowerContent.includes(lowerQuery)) {
            // 提取匹配片段（前后各100字符）
            const index = lowerContent.indexOf(lowerQuery);
            const start = Math.max(0, index - 100);
            const end = Math.min(content.length, index + query.length + 100);
            const snippet = content.substring(start, end);
            
            results.push({
                doc: docName,
                snippet: snippet,
                matchIndex: index
            });
        }
    });
    
    return results;
}
```

```javascript
// 解析Markdown标题
function extractTOC(markdown) {
    const headings = [];
    const regex = /^(#{1,6})\s+(.+)$/gm;
    let match;
    
    while ((match = regex.exec(markdown)) !== null) {
        headings.push({
            level: match[1].length,  // 标题级别 1-6
            text: match[2],          // 标题文本
            id: 'heading-' + headings.length  // 生成ID
        });
    }
    
    return headings;
}

// 生成目录HTML
function renderTOC(headings) {
    let html = '<nav class="toc"><ul>';
    
    headings.forEach(heading => {
        const indent = (heading.level - 1) * 20;  // 缩进
        html += `
            <li style="margin-left: ${indent}px">
                <a href="#${heading.id}">${heading.text}</a>
            </li>
        `;
    });
    
    html += '</ul></nav>';
    return html;
}
```

```javascript
// 保存阅读进度
function saveReadingProgress(docName) {
    const scrollTop = document.getElementById('viewer-content').scrollTop;
    const scrollHeight = document.getElementById('viewer-content').scrollHeight;
    const progress = (scrollTop / scrollHeight * 100).toFixed(1);
    
    localStorage.setItem(`reading_${docName}`, JSON.stringify({
        scrollTop: scrollTop,
        progress: progress,
        timestamp: Date.now()
    }));
}

// 恢复阅读进度
function restoreReadingProgress(docName) {
    const saved = localStorage.getItem(`reading_${docName}`);
    if (saved) {
        const data = JSON.parse(saved);
        document.getElementById('viewer-content').scrollTop = data.scrollTop;
        
        // 显示进度提示
        showProgressToast(`上次读到 ${data.progress}%`);
    }
}
```

#### 方案A：IndexedDB存储（推荐）
```javascript
// 使用IndexedDB存储文档
const dbRequest = indexedDB.open('FitnessKnowledge', 1);

dbRequest.onupgradeneeded = (event) => {
    const db = event.target.result;
    const store = db.createObjectStore('documents', { keyPath: 'name' });
    store.createIndex('category', 'category', { unique: false });
    store.createIndex('tags', 'tags', { multiEntry: true });
};

// 按需加载文档
async function loadDocument(docName) {
    const transaction = db.transaction(['documents'], 'readonly');
    const store = transaction.objectStore('documents');
    const request = store.get(docName);
    
    return new Promise((resolve, reject) => {
        request.onsuccess = () => resolve(request.result);
        request.onerror = () => reject(request.error);
    });
}
```

**优势：**
- ✅ 按需加载，减少初始负载
- ✅ 支持离线缓存
